# Capitalist

	description = "civic_tooltip_capitalist_effects"
	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		always = no
	}
	possible = {
		authority = {
			NOR = {
				value = auth_democratic

			}
		}
	}

}
