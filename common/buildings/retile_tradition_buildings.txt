# Restaurant Megachain
building_restaurant_megachain = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { 
			is_regular_empire = yes
			has_tradition = tr_prosperity_public_works
		}
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}
	
	allow = {
		OR = { 
			retile_urban = yes
			has_modifier = resort_colony
		}
		#retile_ready = yes
		hidden_trigger = { OR = { owner = { is_ai = no } OR = {
			NOT = { is_planet_class = pc_city }
			has_planet_flag = ai_trade_arcology
		} } }
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				owner = { is_ai = yes }
				is_planet_class = pc_city
				NOT = { has_planet_flag = ai_trade_arcology }
			}
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_med
		}	
		upkeep = {
			minerals = 0
			energy = 10
		}		
	}
	
	planet_modifier = {
		pop_growth_speed = 0.10
		#planet_housing_add = 2
		#job_custodian_add = 2
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_restaurant_megachain_effect
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {
			weight = 10000
			owner = { num_owned_planets < 5 }
			count_pops = {
				limit = {  
					has_job = job_clerk
				}
				count > 4
			}
		}
		modifier = {	# Help solve energy problems!!!
			weight = 200000
			owner = { has_monthly_income = { resource = energy value < 25 } }
		}
		modifier = {	# goods security
			weight = 20000
			owner = { has_monthly_income = { resource = consumer_goods value < 25 } }
		}
	}
}
