building_crude_huts = {
	capital = yes
	can_build = no
	can_demolish = yes
	can_be_ruined = no
	
	potential = {
		exists = owner
		owner = { 
			is_primitive = yes
			has_country_flag = stone_age 
		}
	}
	
	destroy_trigger = {
		owner = { is_primitive = no }
	}
	
	planet_modifier = {
		planet_housing_add = 1
		job_hunter_gatherer_add = 1
	}
		
	ai_weight = {
		weight = 1
	}
}
