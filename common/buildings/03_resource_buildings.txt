building_hydroponics_farm = {
	base_buildtime = 480
	base_cap_amount = 1
	category = resource
	can_be_disabled = no
	
	potential = {
		exists = owner
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		owner = { is_ai = no }
	}

	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
			owner = { is_ai = yes }
		}
	}

	prerequisites = {
		"tech_hydroponics"
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_med
		}
		upkeep = {
			energy = 2
		}
	}	
	
	country_modifier = {
		country_resource_max_add = 2000
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_plantoid_empire = no }
		}
		text = retile_food_buff_city_effect
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_plantoid_empire = yes }
		}
		text = retile_food_buff_city_plantoid_effect
	}
	
	ai_weight = {
		weight = 1
	}
}
