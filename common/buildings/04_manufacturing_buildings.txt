building_foundry_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = manufacturing
	can_be_disabled = no
	
	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		retile_urban = yes
		custom_tooltip = {
			fail_text = "retile_requires_engineering_arcology"
			OR = {
				NOT = { is_planet_class = pc_city }
				num_districts = { type = district_engineering_arcology value > 0 }
			}
		}
		#retile_ready = yes
	}
	
	destroy_trigger = {
		retile_urban = no	
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = @cost_low
		}
		upkeep = {
			energy = 10
			#alloys = 1
		}
	}
	
	upgrades = {
		building_foundry_2
	}

	prerequisites = {
		tech_basic_industry
	}	
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_custodian_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_maintenance_drone_add = 1
		}
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_foundry_1_effect
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {	# Safe to develop our empire?
			weight = 150
			retile_basic_economy = yes
		}
		modifier = {	# alloys security
			factor = 10
			OR = {
				owner = { has_monthly_income = { resource = alloys value < 25 } }
				AND = {
					years_passed > 100
					owner = { has_monthly_income = { resource = alloys value < 50 } }
				}
				AND = {
					years_passed > 200
					owner = { has_monthly_income = { resource = alloys value < 100 } }
				}
			}
		}
		modifier = {	# Ministry!!!
			factor = 5
			OR = {
				has_building = building_ministry_production
				has_building = building_production_center
			}
		}
	}

	ai_resource_production = {
		alloys = 1
		trigger = {
			always = yes
		}
	}
}

	triggered_desc = {
