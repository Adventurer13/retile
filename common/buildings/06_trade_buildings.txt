@cost_low = 500
@cost_med = 1000
@cost_high = 1500
@cost_ultra = 2000

#############
# Trade buildings
#############
building_trade_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { 
			is_regular_empire = yes 
			NOT = { has_tradition = tr_prosperity_public_works }
		}
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				owner = { has_country_flag = retile_ai_needs_energy }
				any_owned_pop = { has_job = clerk }
			}
			owner = { has_authority = auth_corporate }
		}
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			owner = { has_tradition = tr_prosperity_public_works }
		}
	}
	
	convert_to = {
		building_bourse_1
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}	
		upkeep = {
			energy = 1
		}		
	}
	
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	
	show_tech_unlock_if = { 
		is_regular_empire = yes
		NOT = { has_tradition = tr_prosperity_public_works }
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_commercial_zone_effect
	}
	
	upgrades = {
		"building_trade_2"
	}
	
	ai_weight = {
		weight = 1
	}
}
