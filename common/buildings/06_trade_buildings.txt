building_trade_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}
	
	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}	
		upkeep = {
			energy = 1
		}		
	}
	
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	
	show_tech_unlock_if = { 
		is_regular_empire = yes
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_commercial_zone_effect
	}
	
	upgrades = {
		"building_trade_2"
	}
	
	ai_weight = {
		weight = 1
	}
}
