@cost_low = 500
@cost_med = 1000
@cost_high = 1500
@cost_ultra = 2000

#############
# Science buildings
#############
building_research_lab_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = research
	can_be_disabled = no
	
	potential = {
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
		NOT = { owner = { has_tradition = tr_harmony_bulwark_of_harmony } }
	}
	
	convert_to = {
		building_metascience_lab_1
	}

	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				retile_ai_planet_healthy_building = yes
				owner = { has_country_flag = retile_ai_needs_science }
				any_owned_pop = { has_job = scholar }
			}
		}
	}

	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony
			owner = { has_tradition = tr_harmony_bulwark_of_harmony }
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}
		upkeep = {
			energy = 1
		}
	}
	
	upgrades = {
		"building_research_lab_2"
	}
	
	prerequisites = {
		"tech_basic_science_lab_1"
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_lab_1_effect
	}

	ai_weight = {
		weight = 1
	}
}
