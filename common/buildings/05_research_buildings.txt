building_research_lab_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = research
	can_be_disabled = no
	
	potential = {
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}
	
	allow = {
		retile_urban = yes
		#retile_ready = yes
	}

	destroy_trigger = {
		OR = {
			retile_urban = no
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = @cost_low
		}
		upkeep = {
			energy = 10
			#alloys = 1
		}
		upkeep = {
			trigger = { owner = { is_hive_empire = yes } }
			energy = -5
			consumer_goods = 1
		}
	}
	
	upgrades = {
		"building_research_lab_2"
	}
	
	prerequisites = {
		"tech_basic_science_lab_1"
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_custodian_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_maintenance_drone_add = 1
		}
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_lab_1_effect
	}

	ai_weight = {
		weight = 200
		
		modifier = {	# Safe to advance our empire?
			weight = 100
			retile_advanced_economy = yes
		}
		modifier = {
			weight = 150000
			is_capital = yes
		}
		modifier = {	# behind in tech!!!
			weight = 100
			owner = {
				any_neighbor_country = { 
					count >= 1
					is_country_type = default
					prev = {
						relative_power = { 
							who = prev
							category = technology 
							value > equivalent
						}
						relative_power = { 
							who = prev 
							category = all 
							value < superior
						}
					}
				}
			}
		}
		modifier = {	# Ministry!!!
			factor = 5
			OR = {
				has_building = building_institute
				has_building = building_supercomputer
			}
		}
		modifier = {	# consumer goods collapse!!!
			factor = -1
			owner = { has_monthly_income = { resource = consumer_goods value < -50 } }
		}
	}
	
	ai_resource_production = {
		physics_research = 1
		society_research = 1
		engineering_research = 1
		trigger = {
			always = yes
		}
	}
}
