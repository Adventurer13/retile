# Abandon a colony
decision_retile_abandon_planet = {
	owned_planets_only = yes

	enactment_time = 360
	
	potential = {
		is_under_colonization = no
		num_pops > 0
	}

	allow = {
		always = yes
		# probaby needs a few more limits, to prevent abuse...
	}
	
	effect = {
		custom_tooltip = decision_retile_abandon_planet_effects
		hidden_effect = {
			random_pop = { retile_abandon_migrate_outside_border = yes }
			if = {
				limit = { num_pops > 0 }
				every_pop = { retile_abandon_migrate = yes }
			}
			retile_abandon_infrastructure = yes
		}
	}
	
	ai_weight = {
		weight = -1000
	}
}
