		potential = {
			is_lithoid = yes
			NOT = { has_trait = trait_hive_mind }

	}
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_slow_breeders
			NOT = { has_trait = trait_hive_mind }
		}
		pop_growth_speed = @pop_surge_trait_penalty
	}
	triggered_planet_modifier = {
		potential = {
			NOT = { has_trait = trait_hive_mind }
			OR = {
				has_trait = trait_fertile
				has_trait = trait_telluric
				has_trait = trait_rapid_breeders
				has_trait = trait_arthropoid_wing
			}
		}
		pop_growth_speed = @pop_surge_trait_bonus
	}
	triggered_planet_modifier = {
		potential = {
			NOT = { has_trait = trait_hive_mind }
			exists = owner
			owner = { has_ascension_perk = ap_engineered_evolution }
		}
		pop_growth_speed = @pop_surge_bonus_cloning
	}
	triggered_planet_modifier = {
		potential = {
			NOT = { has_trait = trait_hive_mind }
			exists = owner
			owner = { has_ethic = ethic_xenophile }
		}
		pop_growth_speed = @pop_surge_bonus_xenophile
	}
	triggered_planet_modifier = {
		potential = {
			NOT = { has_trait = trait_hive_mind }
			exists = owner
			owner = { has_ethic = ethic_fanatic_xenophile }
		}
		pop_growth_speed = @pop_surge_bonus_fanatic_xenophile
	}
	triggered_planet_modifier = {
		potential = {
			NOT = { has_trait = trait_hive_mind }
			exists = owner
			owner = { has_country_flag = relic_last_baol_activated }
		}
		pop_growth_speed = @pop_surge_bonus_baol_relic
	}

	triggered_planet_modifier = {
		potential = { NOT = { has_trait = trait_communal } }
		pop_growth_speed_reduction = @pop_sprawl_trait
	}
	triggered_planet_modifier = {
		potential = { has_trait = trait_solitary }
		pop_growth_speed_reduction = @pop_sprawl_trait
	}
	triggered_planet_modifier = {
		potential = {
			is_organic_species = yes
			exists = planet
			NOT = { planet = { is_planet_class = pc_gaia } }
		}
		pop_growth_speed_reduction = @pop_sprawl_non_gaia
	}
	
	triggered_planet_modifier = {
		potential = {
			is_organic_species = yes
			NOT = { has_trait = trait_hive_mind }
		}
		pop_growth_speed_reduction = @pop_sprawl_base
	}
	
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_hive_mind
			exists = planet
			planet = { NOR = {
				has_building = building_spawning_1
				has_building = building_spawning_2
				has_building = building_spawning_3
			} }
		}
		pop_growth_speed_reduction = @pop_sprawl_hive_0
	}
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_hive_mind
			exists = planet
			planet = { has_building = building_spawning_1 }
		}
		pop_growth_speed_reduction = @pop_sprawl_hive_1
	}
	triggered_planet_modifier = {

			has_trait = trait_hive_mind
			exists = planet
			planet = { has_building = building_spawning_2 }
		}
		pop_growth_speed_reduction = @pop_sprawl_hive_2
	}
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_hive_mind
			exists = planet
			planet = { has_building = building_spawning_3 }
		}
		pop_growth_speed_reduction = @pop_sprawl_hive_3
	}
	########################################	END Pop Sprawl

	triggered_planet_modifier = {	# Dirty Pop!
		potential = { has_trait = trait_repugnant }
		modifier = { planet_amenities_add = 10 }
	}
	triggered_planet_modifier = {	# Clean Pop!
		potential = { has_trait = trait_charismatic }
		modifier = { planet_amenities_add = -10 }
	}
	triggered_planet_modifier = {	# Deviant Pop!
		potential = { has_trait = trait_deviants }
		modifier = { planet_crime_add = 2 }
	}
	triggered_planet_modifier = {	# Conformist Pop!
		potential = { has_trait = trait_conformists }
		modifier = { planet_crime_add = -2 }
	}

	pop_modifier = {
		pop_housing_usage_base = 1
		pop_amenities_usage_no_happiness_base = 0
	}

	# scope: pop to be resettled
	allow_resettlement = {
		always = yes
	}

	resettlement_cost = {
		influence = 10
	}

	unemployment_resources = {
		category = unemployment_resources
	}

	resources = {
		category = pop_category_drones

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_lithoid = no
			}
			food = @living_standard_food_normal
		}

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_lithoid = yes
			}
			minerals = @living_standard_food_normal
		}

		upkeep = {
			trigger = {
				is_robotic_species = yes
			}
			energy = @living_standard_energy_normal
		}

		produces = {
			trigger = {
				is_lithoid = yes
				has_trait = trait_lithoid_scintillating
			}
			rare_crystals = @lithoid_trait_resource_production
		}

		produces = {
			trigger = {
				is_lithoid = yes
				has_trait = trait_lithoid_gaseous_byproducts
			}
			exotic_gases = @lithoid_trait_resource_production
		}

		produces = {
			trigger = {
				is_lithoid = yes
				has_trait = trait_lithoid_volatile_excretions
			}
			volatile_motes = @lithoid_trait_resource_production
		}
		produces = {	### Natural Scientist Traits
			trigger = { 
				OR = { 
					has_trait = trait_natural_physicists
					has_trait = trait_prodigy
				}
			}
			physics_research = 1
		}
		produces = {
			trigger = { 
				OR = { 
					has_trait = trait_natural_sociologists
					has_trait = trait_prodigy
				}
			}
			society_research = 1
		}
		produces = {
			trigger = { 
				OR = { 
					has_trait = trait_natural_engineers
					has_trait = trait_prodigy
				}
			}
			engineering_research = 1
		}				### End Natural Scientist Traits
	}
}
