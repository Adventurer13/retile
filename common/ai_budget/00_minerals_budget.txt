## Expenditure ##
#################
minerals_expenditure_buffer = { # Buffer, will not be spent
	resource = minerals 
	type = expenditure
	category = buffer
	
	potential = {
		always = yes
	}
	
	fraction = {
		weight = 0.05
	}	
}

minerals_expenditure_trade = {
	resource = minerals 
	type = expenditure
	category = trade
	
	potential = {
		has_resource = { 
			type = minerals
			amount > 1000
		}
	}
	
	fraction = {
		weight = 1
	}
}

minerals_expenditure_colonies_lithoid_expand = {
	resource = minerals
	type = expenditure
	category = colonies

	potential = {
		is_machine_empire = no
		is_lithoid_empire = yes
		ai_colonize_plans > 0
		has_resource = { type = minerals amount > 500 }
		has_monthly_income = { resource = minerals value > 0 }
	}

	fraction = {
		weight = 0.5
	}

	static_min = {
		base = 200
	}

	static_max = {
		base = 250
	}
}
