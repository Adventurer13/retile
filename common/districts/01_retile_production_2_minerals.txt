@base_cost = 150
@base_upkeep = 5
@base_buildtime = 240

district_mining = {
	base_buildtime = @base_buildtime
	
	min_for_deposits_on_planet = 1
	max_for_deposits_on_planet = 15
	
	show_on_uncolonized = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	potential = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
		hidden_trigger = {
			retile_ready_minerals = yes
		}
	}

	prerequisites = {
		tech_mechanized_mining
	}

	conversion_ratio = 1
	convert_to = {
		district_mining_uncapped
	}
	
	resources = {
		category = planet_districts
		cost = {
			alloys = @base_cost
		}
		upkeep = {
			trigger = { owner = { is_primitive = no } }
			energy = @base_upkeep
		}		
		upkeep = {
			trigger = { owner = { is_hive_empire = yes } }
			energy = -5
			consumer_goods = 1
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_mining_drone_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_miner_add = 1
		}
	}
	
	# Mining Guilds
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_mining_guilds }
		}
		modifier = {
			planet_housing_add = 1
			job_foundry_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_mining_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		text = job_miner_effect_desc
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {	# minrals security
			weight = 100
			owner = { has_monthly_income = { resource = minerals value < 100 } }
		}
		modifier = {	# minerals emergency!
			weight = 1000
			owner = { has_monthly_income = { resource = minerals value < 25 } }
		}
	}
	
	ai_resource_production = {
		minerals = 1
		trigger = {
			always = yes
		}
	}
}
