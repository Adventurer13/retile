@base_cost = 150
@base_upkeep = 0
@base_buildtime = 240

district_generator = {
	base_buildtime = @base_buildtime

	min_for_deposits_on_planet = 3	
	max_for_deposits_on_planet = 15
	
	show_on_uncolonized = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	potential = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = penal_colony }
		hidden_trigger = {
			retile_ready = yes
		}
	}
	
	prerequisites = {
		tech_power_plant_1
	}

	conversion_ratio = 1
	convert_to = {
		district_generator_uncapped
	}
	
	resources = {
		category = planet_districts
		cost = {
			alloys = @base_cost
		}
		upkeep = {
			trigger = { owner = { is_primitive = no } }
			energy = @base_upkeep
		}
		upkeep = {
			trigger = { owner = { is_hive_empire = yes } }
			consumer_goods = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_technician_drone_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			planet_housing_add = 1
			job_technician_add = 1
		}
	}
	
	# Slave Colony
	triggered_planet_modifier = {
		potential = {
			has_modifier = slave_colony
			owner = { has_ascension_perk = ap_mega_specialization }
		}
		modifier = {
			planet_housing_add = 1
			job_technician_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_technician_effect_desc
	}
	
	ai_weight = {
		weight = 0
		
		modifier = {	# energy security
			weight = 10
			OR = {
				retile_template_good_energy = yes
				AND = {
					owner = { has_monthly_income = { resource = energy value < 100 } }
					num_districts = { type = district_generator value < 2 }
				}
			}
		}
		modifier = {	# energy emergency!
			weight = 1000
			owner = { has_monthly_income = { resource = energy value < 25 } }
		}
		modifier = {	# avoid bad worlds...
			factor = 0.5
			retile_template_good_energy = no
		}
		modifier = {	# avoid urban worlds...
			factor = 0.5
			retile_urban_development = yes
		}
		modifier = {	# greatly prefer worlds with energy buildings
			factor = 10
			has_building = building_energy_grid
			has_building = building_energy_nexus
		}
		modifier = {
			factor = 0
			owner = { has_monthly_income = { resource = energy value > 1000 } }
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}
