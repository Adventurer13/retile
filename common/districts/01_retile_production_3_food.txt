@base_cost = 150
@base_upkeep = 5
@base_buildtime = 240

district_farming = {
	base_buildtime = @base_buildtime

	min_for_deposits_on_planet = 3	
	max_for_deposits_on_planet = 15
	
	show_on_uncolonized = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_relic }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	potential = {
		habitable_structure = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_relic }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = penal_colony }
		hidden_trigger = {
			retile_ready_food = yes
		}
	}

	prerequisites = {
		tech_industrial_farming
	}
	
	conversion_ratio = 1
	convert_to = {
		district_farming_uncapped
	}	
		
	resources = {
		category = planet_districts
		cost = {
			alloys = @base_cost
		}
		upkeep = {
			trigger = { OR = {
				owner = { is_hive_empire = yes }
				AND = {
					NOT = { planet = { has_deposit = retile_deposit_bioreactor } }
					owner = { is_primitive = no }
				}
			} }
			energy = @base_upkeep
		}		
		upkeep = {
			trigger = { owner = { is_hive_empire = yes } }
			energy = -5
			consumer_goods = 1
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_agri_drone_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
			NOT = { has_deposit = retile_deposit_bioreactor }
		}
		modifier = {
			planet_housing_add = 1
			job_farmer_add = 1
		}
	}	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
			has_deposit = retile_deposit_bioreactor
		}
		modifier = {
			planet_housing_add = 1
			job_technician_add = 1
		}
	}	
		
	# Slave Colony
	triggered_planet_modifier = {
		potential = {
			has_modifier = slave_colony
			owner = { has_ascension_perk = ap_mega_specialization }
		}
		modifier = {
			planet_housing_add = 1
			job_farmer_add = 1
		}
	}
		
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_plantoid_empire = no
			}
			NOT = { has_deposit = retile_deposit_bioreactor }
		}
		text = job_farmer_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_plantoid_empire = yes
			}
			NOT = { has_deposit = retile_deposit_bioreactor }
		}
		text = job_farmer_plantoid_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
			has_deposit = retile_deposit_bioreactor
		}
		text = job_technician_effect_desc
	}
	
	ai_weight = {
		weight = 100
		modifier = {	# farming emergency!
			weight = 5000
			OR = {
				AND = {
					country_uses_food = yes
					owner = { has_monthly_income = { resource = food value < 5 } }
				}
				AND = {
					is_plantoid_empire = yes
					habitability = { who = owner.owner_species value < 0.5 }
				}
			}
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
